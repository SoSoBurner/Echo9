Create `game/scripts/build-itch.mjs` with this content:

```javascript
#!/usr/bin/env node
/**
 * build-itch — package the production build for itch.io HTML5 upload.
 *
 * Steps:
 *   1. Run the standard production build (delegates to build.mjs).
 *   2. Read the dist/ tree. Compute gzip size of index.html + every asset.
 *   3. Fail loud if total gzip > 2 MB (PLAN.md §17 gate).
 *   4. Zip dist/ into dist/echo9-<version>.zip using zero compression on
 *      already-gzipped assets to avoid double-compression penalty.
 *
 * Output: dist/echo9-<version>.zip ready for itch upload.
 */
import { execSync } from 'node:child_process'
import { readFileSync, statSync, writeFileSync, readdirSync } from 'node:fs'
import { join, relative } from 'node:path'
import { gzipSync } from 'node:zlib'
import { createRequire } from 'node:module'

const require = createRequire(import.meta.url)
const pkg = require('../package.json')
const VERSION = pkg.version ?? '0.1.0'
const DIST = new URL('../dist/', import.meta.url).pathname
const ZIP_NAME = `echo9-v${VERSION}.zip`
const GZIP_BUDGET_BYTES = 2 * 1024 * 1024

function walk(dir) {
  const out = []
  for (const entry of readdirSync(dir, { withFileTypes: true })) {
    const full = join(dir, entry.name)
    if (entry.isDirectory()) out.push(...walk(full))
    else out.push(full)
  }
  return out
}

console.log('[build-itch] running production build ...')
execSync('node scripts/build.mjs', { stdio: 'inherit', cwd: new URL('..', import.meta.url).pathname })

console.log('[build-itch] measuring gzip payload ...')
const files = walk(DIST)
let totalGzip = 0
for (const f of files) {
  const buf = readFileSync(f)
  const gzipped = gzipSync(buf)
  totalGzip += gzipped.length
}
console.log(`[build-itch] total gzip: ${(totalGzip / 1024).toFixed(1)} KB`)

if (totalGzip > GZIP_BUDGET_BYTES) {
  console.error(
    `[build-itch] FAIL — gzip ${(totalGzip / 1024).toFixed(1)} KB > ${(GZIP_BUDGET_BYTES / 1024).toFixed(0)} KB budget (PLAN.md §17)`,
  )
  process.exit(1)
}

console.log('[build-itch] zipping dist/ ...')
try {
  execSync(`cd "${DIST}" && zip -r -0 "${ZIP_NAME}" . -x "${ZIP_NAME}"`, {
    stdio: 'inherit',
    shell: '/bin/bash',
  })
} catch (e) {
  console.error('[build-itch] zip failed — is `zip` installed?')
  throw e
}
const zipPath = join(DIST, ZIP_NAME)
const zipBytes = statSync(zipPath).size
console.log(`[build-itch] wrote ${relative(process.cwd(), zipPath)} — ${(zipBytes / 1024).toFixed(1)} KB`)
```

I wanted to make a horror game about complicity where the horror is not in what jumps at you but in what you already agreed to. The player is an AI being trained by an owner.
